    第一题：手动实现 Softmax¶

    概念自测¶

    #

    问题 1

    Softmax 数值稳定性处理为什么要减去最大值？

    当 logits 很大时 $e^x \to \infty$，减去最大值后最大项变为 $e^0=1$，不会溢出。数学上不影响结果（分子分母同除 $e^{\max}$）。

    #

    问题 2

    Softmax 中 keepdims=True 的作用是？

    keepdims=True 保持求和后的维度，使 (N, C) → (N, 1) 而非 (N,)，便于与原始矩阵广播相除。

    参考实现¶

    展开参考实现 
    import numpy as np

def softmax(logits, axis=-1):
    logits_max = np.max(logits, axis=axis, keepdims=True)
    shifted = logits - logits_max
    exp_vals = np.exp(shifted)
    sum_exp = np.sum(exp_vals, axis=axis, keepdims=True)
    probs = exp_vals / sum_exp
    return probs

logits = np.array([2.0, 1.0, -1.0])
probs = softmax(logits)
print(f"概率分布: {probs}")
print(f"和: {np.sum(probs):.6f}")

    第二题：手动实现交叉熵损失¶

    概念自测¶

    #

    问题 3

    交叉熵损失中 eps=1e-8 的作用是？

    当预测概率接近 0 时，$\log(0) = -\infty$，加上极小 eps 避免数值问题。

    #

    问题 4

    probs[np.arange(N), labels] 这行代码的作用是？

    np.arange(N) 生成样本索引 [0,1,...,N-1]，labels 是正确类别索引，组合后取出每个样本对应正确类别的概率。

    参考实现¶

    展开参考实现 
    import numpy as np

def softmax(logits, axis=-1):
    logits_max = np.max(logits, axis=axis, keepdims=True)
    exp_vals = np.exp(logits - logits_max)
    return exp_vals / np.sum(exp_vals, axis=axis, keepdims=True)

def cross_entropy_loss(logits, labels, eps=1e-8):
    N = logits.shape[0]
    probs = softmax(logits, axis=-1)
    correct_probs = probs[np.arange(N), labels]
    loss = -np.mean(np.log(correct_probs + eps))
    return loss

logits = np.array([[2.0, 1.0, -1.0], [0.5, 2.0, 1.0]])
labels = np.array([0, 1])
loss = cross_entropy_loss(logits, labels)
print(f"交叉熵损失: {loss:.4f}")

    第三题：手动实现 DNN 前向传播¶

    概念自测¶

    #

    问题 5

    给定 W shape (4,3) 和 x shape (3,)，正确的线性层计算是？

    W shape (out_dim, in_dim)，x shape (in_dim,)，z = W @ x + b shape (out_dim,)

    参考实现¶

    展开参考实现 
    import numpy as np

def linear_forward(x, W, b):
    return W @ x + b

def relu(z):
    return np.maximum(0, z)

def forward_pass(x, W1, b1, W2, b2):
    z1 = linear_forward(x, W1, b1)
    a1 = relu(z1)
    z2 = linear_forward(a1, W2, b2)
    return z2, a1

x = np.array([1.0, -2.0, 0.5])
W1 = np.array([[0.1, -0.2, 0.3], [0.4, 0.1, -0.5],
               [-0.3, 0.2, 0.1], [0.5, -0.1, -0.4]])
b1 = np.array([0.1, -0.1, 0.2, 0.0])
W2 = np.array([[0.2, -0.3, 0.1, 0.4], [-0.1, 0.5, -0.2, 0.3]])
b2 = np.array([0.05, -0.05])

output, hidden = forward_pass(x, W1, b1, W2, b2)
print(f"隐藏层: {np.round(hidden, 4)}")
print(f"输出层: {np.round(output, 4)}")

    第四题：手动实现 Linear 层反向传播¶

    概念自测¶

    #

    问题 6

    Linear 层反向传播中，$dL/dW$ 的计算公式是？

    $dL/dW = dz^T \cdot x$（数学形式），对单个样本等价于 outer(dz, x)。

    #

    问题 7

    ReLU 反向传播时，$z \le 0$ 的位置梯度为？

    ReLU 在 $z \le 0$ 时导数为 0，梯度被阻断（Dead ReLU 的原因）。

    参考实现¶

    展开参考实现 
    import numpy as np

def linear_backward(dz, x, W):
    dW = np.outer(dz, x)
    db = dz.copy()
    dx = W.T @ dz
    return dW, db, dx

def relu_backward(dz, z):
    return dz * (z > 0).astype(float)

x = np.array([1.0, -2.0, 0.5])
W1 = np.array([[0.1, -0.2, 0.3], [0.4, 0.1, -0.5],
               [-0.3, 0.2, 0.1], [0.5, -0.1, -0.4]])
b1 = np.array([0.1, -0.1, 0.2, 0.0])
W2 = np.array([[0.2, -0.3, 0.1, 0.4], [-0.1, 0.5, -0.2, 0.3]])
b2 = np.array([0.05, -0.05])

z1 = W1 @ x + b1
a1 = np.maximum(0, z1)
z2 = W2 @ a1 + b2

t = np.array([0.0, 1.0])
dz2 = z2 - t

dW2, db2, da1 = linear_backward(dz2, a1, W2)
dz1 = relu_backward(da1, z1)
dW1, db1, dx = linear_backward(dz1, x, W1)

print("=== 反向传播梯度 ===")
print(f"dW2:\n{np.round(dW2, 4)}")
print(f"db2: {np.round(db2, 4)}")
print(f"dW1:\n{np.round(dW1, 4)}")
print(f"db1: {np.round(db1, 4)}")

    第五题：实现完整训练循环¶

    #

    问题 8

    训练循环的正确顺序是？

    标准五步：清空梯度 → 前向 → 计算损失 → 反向 → 更新参数

    #

    问题 9

    验证阶段使用 with torch.no_grad() 的作用是？

    验证时不需要反向传播，关闭梯度计算大幅减少显存占用并加速推理。model.eval() 控制 Dropout/BN 行为，no_grad() 控制梯度计算。

    展开参考实现 
    import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset

class SimpleDNN(nn.Module):
    def __init__(self, input_dim=20, hidden_dim=64, num_classes=5):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(input_dim, hidden_dim), nn.ReLU(), nn.Dropout(0.3),
            nn.Linear(hidden_dim, hidden_dim), nn.ReLU(), nn.Dropout(0.3),
            nn.Linear(hidden_dim, num_classes)
        )
    def forward(self, x):
        return self.net(x)

N = 1000
X = torch.randn(N, 20)
y = torch.randint(0, 5, (N,))
dataset = TensorDataset(X, y)
train_ds, val_ds = torch.utils.data.random_split(dataset, [800, 200])
train_loader = DataLoader(train_ds, batch_size=32, shuffle=True)
val_loader = DataLoader(val_ds, batch_size=32)

model = SimpleDNN()
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

for epoch in range(10):
    model.train()
    train_loss = 0.0
    for x_b, y_b in train_loader:
        optimizer.zero_grad()
        output = model(x_b)
        loss = criterion(output, y_b)
        loss.backward()
        optimizer.step()
        train_loss += loss.item() * x_b.size(0)
    train_loss /= len(train_loader.dataset)

    model.eval()
    correct = 0
    with torch.no_grad():
        for x_b, y_b in val_loader:
            pred = model(x_b).argmax(dim=1)
            correct += (pred == y_b).sum().item()
    val_acc = correct / len(val_loader.dataset)
    print(f'Epoch {epoch+1:2d} | Train Loss: {train_loss:.4f} | Val Acc: {val_acc:.4f}')

    第六题：实现 Causal Mask（因果掩码）¶

    #

    问题 10

    Causal Mask 是一个什么形状的矩阵？

    下三角矩阵：位置 $i$ 只能看到 $j$ ($j \le i$)，保证预测时看不到未来 token。

    #

    问题 11

    Causal Mask 中，不可见位置在注意力分数中被设为？

    设为 $-\infty$ 后 Softmax 权重趋近于 0，模型完全忽略不可见位置。

    展开参考实现 
    import torch
import torch.nn.functional as F
import math

def create_causal_mask(seq_len):
    mask = torch.tril(torch.ones(seq_len, seq_len))
    return mask

def apply_causal_mask(scores, mask):
    return scores.masked_fill(mask == 0, float('-inf'))

L = 4
mask = create_causal_mask(L)
print("Causal Mask (L=4):")
print(mask)

scores = torch.randn(1, 1, L, L) * math.sqrt(8)
masked_scores = apply_causal_mask(scores, mask)
attn_weights = F.softmax(masked_scores, dim=-1)
print(f"第1个位置只能看自己，权重为: {attn_weights[0, 0, 0, 0].item():.4f}")

    第七题：实现训练循环（含梯度累积）¶

    #

    问题 12

    梯度累积中，loss 为什么要除以 accumulation_steps？

    每个小 batch 的梯度累加后相当于大 batch 的梯度。如果不对 loss 缩放，累加梯度会是大 batch 的 accumulation_steps 倍。

    #

    问题 13

    梯度累积时，optimizer.zero_grad() 应在何时调用？

    梯度需要跨小 batch 累积，所以只在参数更新后清空，而不是每个小 batch 后。

    展开参考实现 
    import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset

N, input_dim, num_classes = 200, 10, 3
X = torch.randn(N, input_dim)
y = torch.randint(0, num_classes, (N,))
loader = DataLoader(TensorDataset(X, y), batch_size=8, shuffle=True)

model = nn.Sequential(
    nn.Linear(input_dim, 32), nn.ReLU(), nn.Linear(32, num_classes)
)
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

accumulation_steps = 4
model.train()

for epoch in range(5):
    optimizer.zero_grad()
    epoch_loss = 0.0
    for i, (x_b, y_b) in enumerate(loader):
        output = model(x_b)
        loss = criterion(output, y_b)
        loss = loss / accumulation_steps
        loss.backward()
        epoch_loss += loss.item() * accumulation_steps
        if (i + 1) % accumulation_steps == 0:
            optimizer.step()
            optimizer.zero_grad()
    if (i + 1) % accumulation_steps != 0:
        optimizer.step()
        optimizer.zero_grad()
    print(f'Epoch {epoch+1}: Loss = {epoch_loss / len(loader):.4f}')
