第 8 题（10 分）—— 完整代码分析：PoetryModel
class PoetryModel(nn.Module):
    def __init__(self, vocab_size=5000, embed_dim=100, hidden_dim=128):
        super().__init__()
        self.embed = nn.Embedding(vocab_size, embed_dim)
        self.lstm = nn.LSTM(embed_dim, hidden_dim, num_layers=2,
                            batch_first=True, dropout=0.3)
        self.mlp = nn.Sequential(
            nn.Dropout(0.3),
            nn.Linear(hidden_dim, 512),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(512, vocab_size)
        )
        self.log_softmax = nn.LogSoftmax(dim=-1)

    def forward(self, x, is_test=False):
        batch_size = x.size(0)
        embed = self.embed(x)
        h0 = torch.zeros(2, batch_size, 128, device=x.device)
        c0 = torch.zeros(2, batch_size, 128, device=x.device)
        out, _ = self.lstm(embed, (h0, c0))
        if is_test:
            out = out[:, -1, :]
        out = self.mlp(out)
        return self.log_softmax(out)

import numpy as np

def softmax(x):
    e_x = np.exp(x - np.max(x, axis=-1, keepdims=True))
    return e_x / np.sum(e_x, axis=-1, keepdims=True)
